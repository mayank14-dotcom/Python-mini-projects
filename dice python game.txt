import random
import time

# time.sleep is a function or a method in python which is used to create a time interval between two tasks:)

time.sleep(1)
print("welcome to this dice game")
print("This is a 2 player dice game")
print("welcome player 1")
print("welcome player 2")

check = input("if you are a single player than you can play with the computer also(yes/no): ")
time.sleep(1)
if check == "yes":
    rounds = 5
    user_score = 0
    computer_score = 0

    for i in range(1, rounds + 1):
        print("\nRound", i)
        user = random.randint(1, 6)
        computer = random.randint(1, 6)
        print("your dice rolled choice:", user)
        print("now computer's chance:", computer)

        if computer > user:
            print("computer's score is higher in this round")
            time.sleep(1)
            computer_score += 1
        elif user > computer:
            print("user's score is higher in this round")
            time.sleep(1)
            user_score += 1
        else:
            print("both got same score this round")

    print("\nGame Over!")
    print("Total user score:", user_score)
    print("Total computer score:", computer_score)

    if computer_score > user_score:
        print("Computer wins the game!")
        time.sleep(3)
    elif user_score > computer_score:
        print("User wins the game!")
        time.sleep(3)
    else:
        print("The game is a tie!")
        time.sleep(1)
