cart = []

while True:
    print("\n1. Add Item\n2. View Cart\n3. Remove Item\n4. Clear Cart\n5. Exit")
    choice = input("Choose an option: ")

    if choice == '1':
        item = input("Enter item name: ")
        cart.append(item)
        print("Item added.")
    elif choice == '2':
        if not cart:
            print("Cart is empty.")
        else:
            print("Cart items:")
            for i, item in enumerate(cart, 1):
                print(f"{i}. {item}")
    elif choice == '3':
        if not cart:
            print("Cart is already empty.")
        else:
            item = input("Enter item name to remove: ")
            if item in cart:
                cart.remove(item)
                print("Item removed.")
            else:
                print("Item not found in cart.")
    elif choice == '4':
        cart.clear()
        print("Cart cleared.")
    elif choice == '5':
        print("Exiting. Thank you.")
        break
    else:
        print("Invalid choice.")
